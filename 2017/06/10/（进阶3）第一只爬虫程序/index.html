3. 爬虫源码

#编码方式为UTF-8
#coding:utf-8

# 首先导入所需库、包
import requests  #用于发送网络请求
import os  #内置模块，用来创建文件夹、转换路径
from bs4 import BeautifulSoup  #用于解析网页

#浏览器请求头，最好加上，详细见requests官方文档，value值可以随意，下面是模拟win下谷歌浏览器
headers = {'User-Agent':'Mozilla/5.0 (Windows NT 6.1; WOW64) '
                    'AppleWebKit/537.1 (KHTML, like Gecko) '
                    'Chrome/22.0.1207.1 Safari/537.1'}
all_url = 'http://mmjpg.com/'  #要爬的网页地址
start_html = requests.get(all_url,headers=headers) #发送请求
start_html.encoding = 'utf-8'  #注意：强制编码utf-8，否则返回值会有乱码
Soup = BeautifulSoup(start_html.text,'lxml')  #解析网页，lxml为指定解析器

#find()只查找一次，find_all()查找所有符合条件的值，返回一个list
all_a = Soup.find('ul').find_all('a',string = True)  #先查找ul标签，在ul标签下查找含字符串的’a‘标签
for a in all_a:
    title = a.get_text()  #取出a标签中的文本
    path = str(title).strip()  #去掉空格，赋值给path
    os.makedirs(os.path.join('/media/lee/72F2460EF245D74F/Pachong',path))  #创建文件夹，路径要自己选
    os.chdir('/media/lee/72F2460EF245D74F/Pachong/'+path)  #切换到该文件夹中
    print('创建名字叫做', path, '的文件夹')
    href = a['href']  #取出a标签的href属性

    #下面爬取二级页面，相同步骤就不添加备注了
    html = requests.get(href,headers = headers)
    html_Soup = BeautifulSoup(html.text,'lxml')
    '''
    下面这一步比较重要，用于三级页面翻页。可以观察到三级页面翻页的规律：网页后面加'/数字'，不同三级页面可翻页值不同
    通过查看二级网页网页源代码可知，三级页面可翻页的值在<div class=’page‘>标签中倒数第二个<a>标签
    '''
    total_page = html_Soup.find('div',class_ = 'page').find_all('a')[-2].get_text()
    for page in range(1,int(total_page)+1):  #获取三级页面url
    page_url = href + '/' +str(page)

    #爬取三级页面
    img_url = requests.get(page_url,headers = headers)
    #img_url.encoding = 'utf-8'
    img_Soup = BeautifulSoup(img_url.text,'lxml')
    img_url = img_Soup.find('div',class_ = 'content').find('img')['src']

    '''
    下面这一步也要注意，给爬取文件命名，要注意img_url[-6：-4]中的符号会不会对后续创建文件造成影响
    如：包含'/'，就会导致错误
    '''
    name = img_url[-6:-4] +'.jpg'
    print(os.getcwd())  #打印当前路径
    print('开始保存', name, '...')
    img = requests.get(img_url,headers = headers)  #找到所需文件地址，发送请求


    #保存文件
    f = open(os.getcwd() + '/' + name, 'ab')  #打开二进制文件，若没有则创建
    f.write(img.content)  #写入二进制响应内容，非文本请求要用content
    f.close()  #关闭文档，否则会占用内存
    print('已下载'+ name)
print('数据采集结束！')
